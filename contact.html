import nodemailer from "nodemailer"; export async function handler(event) { if (event.httpMethod !== "POST") { return { statusCode: 405, body: "Method Not Allowed" }; } try { const { name, email, message } = JSON.parse(event.body); // Configure transporter let transporter = nodemailer.createTransport({ host: "smtp.gmail.com", // Change if using Outlook/Yahoo port: 587, secure: false, auth: { user: process.env.SMTP_USER, // Your Gmail or SMTP username pass: process.env.SMTP_PASS // Your app password or SMTP password } }); // Send email await transporter.sendMail({ from: `"Website Contact" <${process.env.SMTP_USER}>`, to: "youremail@example.com", // Replace with your destination email subject: `New Contact Form Submission from ${name}`, text: `Name: ${name}\nEmail: ${email}\n\nMessage:\n${message}`, html: `

Name: ${name}

Email: ${email}

Message:
${message}

` }); return { statusCode: 200, body: JSON.stringify({ success: true }) }; } catch (error) { console.error("Email sending error:", error); return { statusCode: 500, body: JSON.stringify({ success: false, error: error.message }) }; } }
